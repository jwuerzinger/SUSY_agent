% ----- 12. Unexpected findings ----------------------------
\begin{frame}{Three Unexpected Blind Spots}
\small
\begin{enumerate}
  \item \textbf{Light sleptons with Bino LSP (21 models):}
    Sleptons at 270--580\GeV invisible despite ATLAS reach $\sim$700\GeV.
    pMSSM BRs dilute the clean dilepton signal; SModelS lacks this topology.

  \item \textbf{Compressed stops with displaced charginos (2 models, up to 14\pb):}
    Stops within T2tt contour but Wino LSP creates long-lived $\chipm$ ($\ctau \sim 1$--10\,cm).
    Falls between prompt and displaced searches --- a \textbf{novel gap}.

  \item \textbf{534 Wino models below disappearing-track limit:}
    ATLAS-SUSY-2018-19 excludes pure Winos to $\sim$660\GeV, but all invisible in SModelS.
    Encoding this \textbf{single analysis} would reclassify most ``invisible'' models.
\end{enumerate}
\vspace{0.3em}
\begin{center}
\fbox{\parbox{0.85\textwidth}{\centering
Models at masses and $\sigma$ where simplified-model results predict exclusion --- the pMSSM reveals the gap.
}}
\end{center}
\end{frame}

% ----- 14. Recommendations --------------------------------
\begin{frame}{Recommendations for ATLAS}
\begin{enumerate}
  \item \textbf{Encode existing searches in SModelS} (highest impact):\\
    {\small Add SUSY-2018-19 (disapp.\ tracks), SUSY-2018-32 (sleptons), SUSY-2019-09 (compressed Higgsino).
    Would reclassify the majority of 1,372 ``invisible'' models.}

  \item \textbf{Extend disappearing-track search (Gap A):}\\
    {\small Reach $\sim$460 $\to$ 800+\GeV with ITk. 102 models excludable.}

  \item \textbf{Lower soft-lepton thresholds (Gap B):}\\
    {\small ISR-boosted with \mll binning to 1\GeV. 42 models excludable.}

  \item \textbf{Extend slepton reach (Gap C):}\\
    {\small \mTtwo-based to 400--500\GeV. 165 models excludable.}

  \item \textbf{Novel displaced $b$-jet search (Gap D):}\\
    {\small Displaced vertex + $b$-tagging. $\sigma$ up to 14\pb.}

  \item \textbf{Multi-lepton cascade search (Gap E):}\\
    {\small Asymmetric final-state SRs. 22 models excludable.}
\end{enumerate}
\end{frame}

% ----- 15. Summary ----------------------------------------
\begin{frame}{Summary}
\begin{itemize}
  \item \textbf{2,087 pMSSM models} survive all current constraints (from 26,048 evaluated)
  \item Viable pMSSM dominated by \textbf{compressed electroweakino spectra}: Wino 48\%, Higgsino 48\%, Bino 4\%
  \item \textbf{69.7\% invisible to ATLAS} in SModelS --- but much of this is a SModelS encoding gap
  \item \textbf{Five blind spots} identified with concrete Run~3 search proposals and 30 benchmarks
  \item \textbf{15 truly dark models} invisible to both ATLAS and direct detection (DARWIN)
  \item Narrow corridors (Bino co-annihilation, compressed stop) intrinsically difficult --- importance sampling is the best tool
  \item \textbf{Highest-impact action:} Encode 3 missing ATLAS analyses in SModelS
\end{itemize}
